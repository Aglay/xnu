${MIG_UUHDRS} : \
	%.h : %.defs
	@echo MIG $@
	$(_v)$(MIG) $(MIGFLAGS)		\
		-server /dev/null	\
		-user /dev/null		\
		-header $@ 		\
		$<

${MIG_USHDRS} : \
	%_server.h : %.defs
	@echo MIG $@
	$(_v)$(MIG) $(MIGFLAGS)		\
		-server /dev/null	\
		-user /dev/null		\
		-header /dev/null	\
		-sheader $@ 		\
		$<
