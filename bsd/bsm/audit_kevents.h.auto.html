/*
 * The reserved event numbers for kernel events are 1...2047 and 43001..44900.
 */
#define	AUE_IS_A_KEVENT(e)	(((e) > 0 && (e) < 2048) || 	\
    				 ((e) > 43000 && (e) < 44901))
