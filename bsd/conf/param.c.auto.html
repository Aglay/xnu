unsigned int nmbclusters = NMBCLUSTERS;
int	nport = NPROC / 2;

/*

 */

int aio_max_requests_per_process = CONFIG_AIO_PROCESS_MAX;
int aio_worker_threads = CONFIG_AIO_THREAD_COUNT;
