/*
 * log benchmark - should do wider range...
 */

#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <math.h>

#include "libmicro.h"

int
benchmark_init()
{
	(void) sprintf(lm_usage, "note: measures log()");
	lm_nsecs_per_op = 75;
	lm_tsdsize = 0;
	return (0);
}

/*ARGSUSED*/
int
benchmark(void *tsd, result_t *res)
{
	int			i;
	/* Added as part of the fix for radar 7508837 */
        double                  t = 0.0;

	for (i = 0; i < lm_optB; i += 10) {
		double value = i + .01;
#if 1 /* Apple added code, see radar 7508837 */
                t += log(value);
                t += log(value + 1.0);
                t += log(value + 2.0);
                t += log(value + 3.0);
                t += log(value + 4.0);
                t += log(value + 5.0);
                t += log(value + 6.0);
                t += log(value + 7.0);
                t += log(value + 8.0);
                t += log(value + 9.0);
        }
        res->re_count = i;

        return ((int)(t - t));
#else
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
		(void) log(value);
	}
	res->re_count = i;

	return (0);
#endif /* end of Apple fix  */
}
